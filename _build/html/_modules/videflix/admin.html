class CustomUserAdmin(ImportExportModelAdmin, UserAdmin):
    """
    Customizes the admin interface for the CustomUser model.
    
    Features:
        - list_display: Displays email, is_staff, and is_active in the admin list view.
        - list_filter: Filters users based on staff status, active status, and superuser status.
        - fieldsets: Defines editable fields in the user detail form (email, password, permissions, dates).
        - add_fieldsets: Adds fields for creating new users (email, password1, password2, is_staff, is_active).
        - search_fields: Enables searching by email.
        - ordering: Orders the users by email.
    """
    list_display = ('email', 'is_staff', 'is_active')
    list_filter = ('is_staff', 'is_active', 'is_superuser')

    fieldsets = (
        (None, {'fields': ('email', 'password')}),
        (_('Permissions'), {
            'fields': ('is_active', 'is_staff', 'is_superuser', 'groups', 'user_permissions'),
        }),
        (_('Important dates'), {'fields': ('last_login', 'date_joined')}),
    )

    add_fieldsets = (
        (None, {
            'classes': ('wide',),
            'fields': ('email', 'password1', 'password2', 'is_staff', 'is_active'),
        }),
    )

    search_fields = ('email',)
    ordering = ('email',)




admin.site.register(CustomUser, CustomUserAdmin)
